# 浅拷贝

```python
import copy

a = [1, 2]
b = [2, 3]

c = [a, b]
d = copy.copy(c)

print(a)  # >>> [1, 2]
print(b)  # >>> [2, 3]
print(c)  # >>> [[1, 2], [2, 3]]
print(d)  # >>> [[1, 2], [2, 3]]

a.append([3, 4, 5])
print(a)  # >>> [1, 2, [3, 4, 5]]
print(b)  # >>> [2, 3]
print(c)  # >>> [[1, 2, [3, 4, 5]], [2, 3]]
print(d)  # >>> [[1, 2, [3, 4, 5]], [2, 3]]
```
